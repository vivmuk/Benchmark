// ============ SETUP ============
const canvas = document.getElementById('game');
const ctx = canvas.getContext('2d');

const STATE = { START: 0, PLAYING: 1, PAUSED: 2, GAME_OVER: 3, VICTORY: 4, LEVEL_COMPLETE: 5, BALL_WAIT: 6 };
let gameState = STATE.START;
let score = 0, lives = 3, level = 1;
const MAX_LEVELS = 5;

// ============ AUDIO (Web Audio API synthesized) ============
let audioCtx = null;
function initAudio() {
  if (!audioCtx) audioCtx = new (window.AudioContext || window.webkitAudioContext)();
}
function playSound(type) {
  if (!audioCtx) return;
  const now = audioCtx.currentTime;
  const make = (waveType, freqStart, freqEnd, dur, vol) => {
    const osc = audioCtx.createOscillator();
    const gain = audioCtx.createGain();
    osc.type = waveType;
    osc.frequency.setValueAtTime(freqStart, now);
    if (freqEnd !== freqStart) osc.frequency.exponentialRampToValueAtTime(Math.max(1, freqEnd), now + dur);
    gain.gain.setValueAtTime(vol, now);
    gain.gain.exponentialRampToValueAtTime(0.001, now + dur);
    osc.connect(gain).connect(audioCtx.destination);
    osc.start(now); osc.stop(now + dur);
  };
  switch(type) {
    case 'hit': make('square', 800, 400, 0.1, 0.15); break;
    case 'paddle': make('triangle', 200, 150, 0.08, 0.2); break;
    case 'wall': make('sine', 300, 300, 0.05, 0.08); break;
    case 'lose': make('sawtooth', 400, 80, 0.5, 0.2); break;
    case 'break': {
      const buf = audioCtx.createBuffer(1, audioCtx.sampleRate * 0.15, audioCtx.sampleRate);
      const data = buf.getChannelData(0);
      for (let i = 0; i < data.length; i++) data[i] = (Math.random()*2-1) * Math.pow(1 - i/data.length, 2);
      const noise = audioCtx.createBufferSource(); noise.buffer = buf;
      const filter = audioCtx.createBiquadFilter(); filter.type = 'bandpass'; filter.frequency.value = 1000;
      const gain = audioCtx.createGain();
      gain.gain.setValueAtTime(0.2, now);
      gain.gain.exponentialRampToValueAtTime(0.001, now + 0.15);
      noise.connect(filter).connect(gain).connect(audioCtx.destination);
      noise.start(now); break;
    }
    case 'levelup': [400,500,600,800].forEach((f,i)=>make('square',f,f,0.15,0.15)); break;
    case 'gameover': [400,300,200,100].forEach((f,i)=>{ const t=now+i*0.15; const o=audioCtx.createOscillator(),g=audioCtx.createGain(); o.type='sawtooth';o.frequency.value=f;g.gain.setValueAtTime(0.2,t);g.gain.exponentialRampToValueAtTime(0.001,t+0.2);o.connect(g).connect(audioCtx.destination);o.start(t);o.stop(t+0.2);}); break;
    case 'victory': [523,659,784,1047,1319].forEach((f,i)=>{ const t=now+i*0.12; const o=audioCtx.createOscillator(),g=audioCtx.createGain(); o.type='triangle';o.frequency.value=f;g.gain.setValueAtTime(0.2,t);g.gain.exponentialRampToValueAtTime(0.001,t+0.3);o.connect(g).connect(audioCtx.destination);o.start(t);o.stop(t+0.3);}); break;
  }
}

// ============ ENTITIES ============
const ball = { x:0, y:0, vx:0, vy:0, radius:8, speed:6, trail:[] };
const paddle = { x:0, y:0, width:120, height:14, speed:9 };
let bricks = [], particles = [], stars = [];

const BRICK_COLORS = [
  { fill:'#ff0066', glow:'#ff0066', hp:1, score:10 },
  { fill:'#ff8800', glow:'#ff8800', hp:1, score:15 },
  { fill:'#ffee00', glow:'#ffee00', hp:1, score:20 },
  { fill:'#00ff88', glow:'#00ff88', hp:1, score:25 },
  { fill:'#00f0ff', glow:'#00f0ff', hp:1, score:30 },

];

// ============ INPUT ============
const keys = {};
let mouseX = null, touchX = null, usingMouse = false;
window.addEventListener('keydown', e => {
  keys[e.key.toLowerCase()] = true;
  if (e.key === ' ') { e.preventDefault(); initAudio(); if (gameState === STATE.BALL_WAIT) launchBall(); }
  if (e.key.toLowerCase() === 'p' && (gameState === STATE.PLAYING || gameState === STATE.PAUSED)) togglePause();
});
window.addEventListener('keyup', e => { keys[e.key.toLowerCase()] = false; });
canvas.addEventListener('mousemove', e => {
  const r = canvas.getBoundingClientRect();
  mouseX = (e.clientX - r.left) * (canvas.width / r.width);
  usingMouse = true;
});
canvas.addEventListener('click', () => { initAudio(); if (gameState === STATE.BALL_WAIT) launchBall(); });
canvas.addEventListener('touchmove', e => {
  e.preventDefault();
  const r = canvas.getBoundingClientRect();
  touchX = (e.touches[0].clientX - r.left) * (canvas.width / r.width);
  usingMouse = false;
}, { passive: false });
canvas.addEventListener('touchstart', e => {
  e.preventDefault(); initAudio();
  const r = canvas.getBoundingClientRect();
  touchX = (e.touches[0].clientX - r.left) * (canvas.width / r.width);
  if (gameState === STATE.BALL_WAIT) launchBall();
}, { passive: false });

// ============ RESIZE ============
function resize() {
  const c = document.getElementById('gameContainer');
  const w = c.clientWidth, h = c.clientHeight;
  const aspect = 4/3;
  let cw, ch;
  if (w/h > aspect) { ch = h * 0.95; cw = ch * aspect; }
  else { cw = w * 0.95; ch = cw / aspect; }
  canvas.style.width = cw + 'px';
  canvas.style.height = ch + 'px';
  const dpr = window.devicePixelRatio || 1;
  canvas.width = Math.floor(cw * dpr);
  canvas.height = Math.floor(ch * dpr);
  GW = canvas.width; GH = canvas.height;
  const scale = GW / 800;
  ball.radius = 8 * scale;
  paddle.width = 120 * scale;
  paddle.height = 14 * scale;
  paddle.speed = 9 * scale;
  ball.speed = 6 * scale;
  if (gameState === STATE.START || gameState === STATE.BALL_WAIT) {
    resetPaddle();
    if (gameState === STATE.BALL_WAIT) placeBallOnPaddle();
  }
  initStars();
}
window.addEventListener('resize', resize);

// ============ STARS ============
function initStars() {
  stars = [];
  for (let i = 0; i < 60; i++) {
    stars.push({ x: Math.random()*GW, y: Math.random()*GH, size: Math.random()*1.5+0.3, speed: Math.random()*0.3+0.1, twinkle: Math.random()*Math.PI*2 });
  }
}

// ============ LEVELS ============
function buildLevel(lvl) {
  bricks = [];
  const scale = GW / 800;
  const cols = 11;
  const rows = Math.min(4 + lvl, 8);
  const padding = 6 * scale;
  const offsetTop = 70 * scale;
  const offsetSide = 30 * scale;
  const brickW = (GW - offsetSide*2 - padding*(cols-1)) / cols;
  const brickH = 22 * scale;
  for (let r = 0; r < rows; r++) {
    for (let c = 0; c < cols; c++) {
      let include = true;
      if (lvl === 2) {
        const mid = (cols-1)/2;
        if (Math.abs(c - mid) > rows - r - 1 && r < rows/2) include = false;
      } else if (lvl === 3) {
        if ((r + c) % 2 === 1 && r > 1) include = false;
      } else if (lvl === 4) {
        const mid = Math.floor(cols/2);
        if (c !== mid && r !== Math.floor(rows/2) && r > 2 && Math.abs(c - mid) > 2) include = false;
      }
      if (!include) continue;
      const colorIdx = Math.min(r, BRICK_COLORS.length - 1);
      const color = BRICK_COLORS[colorIdx];
      let hp = color.hp;
      if (lvl >= 4 && r === 0) hp = 2;
      bricks.push({
        x: offsetSide + c*(brickW+padding), y: offsetTop + r*(brickH+padding),
        w: brickW, h: brickH, hp, maxHp: hp, color: color.fill, glow: color.glow, score: color.score, alive: true
      });
    }
  }
}

// ============ RESET / FLOW ============
function resetPaddle() { paddle.x = GW/2 - paddle.width/2; paddle.y = GH - 40; }
function placeBallOnPaddle() { ball.x = paddle.x + paddle.width/2; ball.y = paddle.y - ball.radius - 2; ball.vx = 0; ball.vy = 0; ball.trail = []; }
function launchBall() {
  const angle = (-Math.PI/2) + (Math.random()-0.5)*0.6;
  ball.vx = Math.cos(angle) * ball.speed;
  ball.vy = Math.sin(angle) * ball.speed;
  gameState = STATE.PLAYING;
}
function startGame() {
  initAudio();
  score = 0; lives = 3; level = 1;
  buildLevel(level); resetPaddle(); placeBallOnPaddle();
  gameState = STATE.BALL_WAIT;
  updateHUD(); hideAllOverlays();
  document.getElementById('hud').classList.remove('hidden');
}
function nextLevel() {
  level++;
  if (level > MAX_LEVELS) {
    gameState = STATE.VICTORY; playSound('victory');
    document.getElementById('victoryScore').textContent = score;
    showOverlay('victoryScreen'); return;
  }
  buildLevel(level); resetPaddle(); placeBallOnPaddle();
  gameState = STATE.BALL_WAIT; updateHUD(); hideAllOverlays();
}
function loseLife() {
  lives--; playSound('lose'); updateHUD();
  if (lives <= 0) gameOver();
  else { resetPaddle(); placeBallOnPaddle(); gameState = STATE.BALL_WAIT; }
}
function gameOver() {
  gameState = STATE.GAME_OVER; playSound('gameover');
  document.getElementById('finalScore').textContent = score;
  document.getElementById('finalLevel').textContent = level;
  showOverlay('gameOverScreen');
}
function levelComplete() {
  gameState = STATE.LEVEL_COMPLETE;
  const bonus = lives * 100 + level * 50;
  score += bonus;
  document.getElementById('lvlScore').textContent = score;
  document.getElementById('lvlBonus').textContent = bonus;
  playSound('levelup'); updateHUD();
  showOverlay('levelCompleteScreen');
}
function togglePause() {
  if (gameState === STATE.PLAYING) { gameState = STATE.PAUSED; showOverlay('pauseScreen'); }
  else if (gameState === STATE.PAUSED) { gameState = STATE.PLAYING; hideAllOverlays(); }
}

// ============ UI HELPERS ============
function showOverlay(id) { hideAllOverlays(); document.getElementById(id).classList.remove('hidden'); }
function hideAllOverlays() { document.querySelectorAll('.overlay').forEach(o => o.classList.add('hidden')); }
function updateHUD() {
  document.querySelector('#scoreDisplay .hud-value').textContent = score;
  document.querySelector('#livesDisplay .hud-value').textContent = lives;
  document.querySelector('#levelDisplay .hud-value').textContent = level;
}

// ============ PARTICLES ============
function spawnParticles(x, y, color, count = 15) {
  for (let i = 0; i < count; i++) {
    const angle = Math.random() * Math.PI * 2;
    const speed = Math.random() * 4 + 1;
    particles.push({ x, y, vx: Math.cos(angle)*speed, vy: Math.sin(angle)*speed, life: 1, decay: Math.random()*0.02+0.015, size: Math.random()*3+1, color });
  }
}
function updateParticles() {
  for (let i = particles.length - 1; i >= 0; i--) {
    const p = particles[i];
    p.x += p.vx; p.y += p
